    const getSummary = useCallback((): PolishSummary | null => {
        const str = localStorage.getItem('polishSummary');
        if (!str) return null;
        try {
            return JSON.parse(str);
        } catch (err) {
            console.log("Could not parse summary from localStorage.", err);
            return null;
        }
